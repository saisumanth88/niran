on-prem
---------
create user ggadmin identified by Ggadmin_Prd_0nPrem default tablespace users temporary tablespace temp;
grant connect, resource to ggadmin;
grant select any dictionary to ggadmin;
grant create view to ggadmin;
grant execute on dbms_lock to ggadmin;
exec dbms_goldengate_auth.GRANT_ADMIN_PRIVILEGE('ggadmin');
ALTER USER ggadmin QUOTA UNLIMITED ON USERS;


RDS

CREATE TABLESPACE goldengate_idx;
CREATE USER ggadmin IDENTIFIED BY "Ggadmin_rds_Pr0" DEFAULT TABLESPACE goldengate TEMPORARY TABLESPACE temp;
ALTER USER ggadmin QUOTA UNLIMITED ON goldengate;
ALTER USER ggadmin QUOTA UNLIMITED ON goldengate_idx;
